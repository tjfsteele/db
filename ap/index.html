function updatePrice() { document.getElementById('time').innerHTML = "Waking up Wi-Fi..."; // Give the Kobo 5 seconds to stabilize its connection after the refresh setTimeout(function() { var url = "https://script.google.com/macros/s/AKfycby4gf4zULR-cQzekmZpXoBs5txrVYr7TJIQWCcwfbwkVy1uqrpz1aniyvkweHTfRbue/exec?cache=" + new Date().getTime(); var xhr = new XMLHttpRequest(); document.getElementById('time').innerHTML = "Fetching data..."; xhr.open("GET", url, true); xhr.timeout = 20000; // 20 second timeout xhr.onreadystatechange = function () { if (xhr.readyState == 4) { if (xhr.status == 200) { var parts = xhr.responseText.split("|"); document.getElementById('price').innerHTML = parts[0]; document.getElementById('next-price').innerHTML = parts[1]; var now = new Date(); var h = now.getHours(); var m = now.getMinutes(); document.getElementById('time').innerHTML = "Updated: " + (h<10?'0':'')+h + ":" + (m<10?'0':'')+m; } else { document.getElementById('time').innerHTML = "Error: Status " + xhr.status; } } }; xhr.onerror = function() { document.getElementById('time').innerHTML = "Network Error - Retrying..."; setTimeout(updatePrice, 5000); // Try again in 5 seconds }; xhr.send(); }, 5000); } // Start the sequence updatePrice();
